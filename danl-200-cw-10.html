2 Q1a

  • Describe how the distribution of rating varies across week 1, week 2, and week 3 using the faceted histogram.
q1a <- billboard %>% 
  pivot_longer(cols = wk1:wk76,
               names_to = "week",
               values_to = "rating") %>% 
  filter(week %in% c('wk1', 'wk2', 'wk3'))
ggplot(q1a,
       aes(x = rating)) +
  geom_histogram() +
  facet_wrap(.~ week)
